import java.util.PriorityQueue;

class Solution {
    public int[] solution(int k, int[] score) {
        int[] answer = new int[score.length];
        PriorityQueue<Integer> q = new PriorityQueue<Integer>();
        for(int i=0; i<score.length; i++){
            q.add(score[i]);
            if(q.size()>k)q.remove();
            answer[i] = q.peek();
           }
        return answer;
    }
}
